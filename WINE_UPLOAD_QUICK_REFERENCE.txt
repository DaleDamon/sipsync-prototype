REQUIRED FIELDS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. NAME (Wine name)
   Example: "Cabernet Sauvignon 2019"

2. TYPE (Must be exactly one)
   ├─ red
   ├─ white
   ├─ rosé
   ├─ sparkling
   └─ dessert

3. PRICE (USD, numbers only)
   Examples: 18, 45, 65, 350
   ✓ 65        ✓ 18.50
   ✗ $65       ✗ 65.00 USD

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
OPTIONAL FIELDS (Leave blank if unknown)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

ACIDITY
  ├─ low      (soft, smooth)
  ├─ medium   (balanced)
  └─ high     (bright, crisp)

TANNINS
  ├─ low      (smooth, silky)
  ├─ medium   (balanced)
  └─ high     (dry, gripping - mostly reds)

BODY WEIGHT
  ├─ light    (thin, delicate)
  ├─ medium   (balanced)
  └─ full     (heavy, rich)

SWEETNESS LEVEL
  ├─ dry      (no residual sugar)
  ├─ medium   (slightly sweet)
  └─ sweet    (dessert-like)

FLAVOR PROFILE (choose multiple, comma-separated)
  oak, cherry, citrus, berry, vanilla, spice, floral, chocolate,
  earthy, tropical, herbal, honey

UPLOAD STEPS
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Download template: RESTAURANT_WINE_MENU_TEMPLATE.csv

2. Open in Excel or Google Sheets

3. Fill in your wines:
   - Row per wine
   - Match column headers exactly

4. Validate data:
   ✓ All wines have name, type, price
   ✓ Type is: red, white, rosé, sparkling, or dessert
   ✓ Price is a number (no $)
   ✓ Optional fields match allowed values
   ✓ Flavors are in approved list

5. Export as CSV:
   Excel: File → Save As → CSV UTF-8
   Google Sheets: File → Download → CSV

6. Upload via API or contact support
